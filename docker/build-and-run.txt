# Building Image
docker build --build-arg CACHE_DATE=$(date +%Y-%m-%d:%H:%M:%S) -t dotfiles-tester:latest .

# Running dotfiles-tester
docker run -it dotfiles-tester /bin/bash

# Testing dotfiles
cd dotfiles
./bootstrap.sh

# Reload bash
exec bash

# Docker container/image cleanup
docker rm $(docker ps -a -q)
docker image prune
